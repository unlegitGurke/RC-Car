function createGauge(name, label, min, max, sizebias, gx, gy) {
   // Konfigurationsobjekt für das Messgerät
   var config = {
       size: 200 + sizebias,

       coorx: gx,

   };
 
   // Berechnet den Wertebereich des Messgeräts
   var range = config.max - config.min;
 
   // Erstellt ein neues Messgerät und rendert es
   gauges[name] = new Gauge(name + "GaugeContainer", config);

// Funktion zum Erstellen der Pfade für die Positionierung der Messgeräte
function createPaths() {

// Funktion zum Aktualisieren der Messgeräte
function updateGauges() {
   // Durchläuft alle Messgeräte im Array
   for (var key in gauges) {
       // Generiert einen zufälligen Wert für das Messgerät
       var value = getRandomValue(gauges[key]);
       // Aktualisiert das Messgerät mit dem neuen Wert
       gauges[key].redraw(value);
   }
}
 
// Funktion zum Generieren eines zufälligen Werts für ein Messgerät
function getRandomValue(gauge) {
   var overflow = 0; //10;
   return Math.floor(gauge.config.min - overflow + (gauge.config.max - gauge.config.min + overflow * 2) * Math.random());
}

       this.config.coory = configuration.coory;
       this.config.cx = this.config.size / 2;
       this.config.cy = this.config.size / 2;
       this.config.inner = this.config.size / (25 / 7);
       this.config.outer = this.config.size / (25 / 9.5);
   };
 
   // Methode zum Rendern des Messgeräts
   this.render = function () {
       // Funktion zum Erstellen eines Bogen-Generators für das Messgerät
       var arc = d3.svg.arc()
           .innerRadius(this.config.inner)
           .outerRadius(this.config.outer)
           .startAngle(0);
 
       // Erstellt das Container-Div für das Messgerät
       this.container = d3.select("body")
           .append("div")
           .attr("id", this.config.label + "Temp")
           .attr("class", "tempContainer")
           .style("left", this.config.coorx + "px")
           .style("top", this.config.coory + "px")
           .style("position", "absolute");
 
       // Erstellt das SVG-Element für das Messgerät
       this.body = this.container
           .append("svg")
           .attr("width", this.config.size)
           .attr("height", this.config.size);

       var defs = this.body.append("defs");
       var filter = defs.append("filter")
           .attr("id", "drop-shadow")
           .attr("height", "130%");

           .attr("in", "blur")
           .attr("dx", 0)
           .attr("dy", 0)

       feMerge.append("feMergeNode")
           .attr("in", "offsetBlur");
       feMerge.append("feMergeNode")
           .attr("in", "SourceGraphic");

       var pointerContainer = this.body.append("g")
           .attr("class", "pointerContainer")

           .style("fill", "#565656")
           .attr("d", arc);

           .datum({ endAngle: this.valueToRadians(this.config.min) })
           .style("fill", "#ff5f24")
           .attr("class", "foreground")
           .attr("d", arc);
 
       // Erstellt den Körper des Messgeräts
       pointerContainer
           .append("circle")
           .style("fill", "#e6e7e8")
           .style("class", "gaugebody")
           .style("filter", "url(#drop-shadow)")
           .attr("r", this.config.inner);
 
       // Berechnet den mittleren Wert des Messgeräts
       var midValue = (this.config.min + this.config.max) / 2;
 
       // Berechnet die Schriftgröße basierend auf der Größe des Messgeräts
       var fontSize = Math.round(this.config.size / 15);
 
       // Fügt den Text für die aktuelle Temperatur hinzu
       pointerContainer
           .append("text")
           .attr("x", this.config.cx - (this.config.cx * 1.0))
           .attr("y", this.config.cy - (this.config.cy * 1.20))
           .attr("text-anchor", "middle")
           .attr("fill", "#58585b")
           .style("font-size", fontSize + "px")
           .text('Current Temp.');
 
       // Aktualisiert die Schriftgröße für den aktuellen Temperaturwert
       fontSize = Math.round(this.config.size / 3.5);
       var current = pointerContainer
           .append("text")
           .attr("x", this.config.cx - (this.config.cx * 0.92))
           .attr("y", this.config.cy - (this.config.cy * 0.7))
           .attr("class", "current")
           .attr("text-anchor", "middle")
           .attr("fill", "#58585b")
           .style("font-size", fontSize + "px")
           .text(this.config.min + degreeSign);
 
       // Ruft die redraw-Methode auf, um das Messgerät zu aktualisieren
       this.redraw(this.config.min, 0);
   };
 
   // Methode zum Aktualisieren des Messgeräts
   this.redraw = function (value, transitionDuration) {
       // Konvertiert den Wert in Grad und Bogenmaß
       var degreeValue = this.valueToDegrees(value);
       var pointerValue = this.valueToRadians(value);
       redrawSize = this.config.size;
 
       // Bogen-Generator für das Messgerät
       var arc = d3.svg.arc()
           .innerRadius(this.config.inner)
           .outerRadius(this.config.outer)
           .startAngle(0);
 
       // Bestimmt die Farbe des Messgeräts basierend auf dem Wert
       switch (true) {
           case (degreeValue >= 270):
               // Dunkelrot
               newColor = '#990000';
               break;
           case (degreeValue >= 180):
               // Hellblau
               newColor = '#ff5f24';
               break;
           case (degreeValue >= 90):
               // Hellrot
               newColor = '#009eff';
               break;
           default:
               // Dunkelblau
               newColor = '#065aa9';
               break;
       }
 
       // Aktualisiert den Text für die aktuelle Temperatur
       var current = this.body.select(".pointerContainer");
       current.select(".current")
           .transition()
           .text(value + degreeSign);
 
       // Aktualisiert die Farbe des Vordergrunds des Messgeräts mit Übergang
       foreground = this.body.select(".pointerContainer");
       foreground.select(".foreground")
           .transition()
           .duration(750)
           .styleTween("fill", function () {
               return d3.interpolate(newColor, defaultColor);
           })
           .call(arcTween, pointerValue);
 
       // Speichert die Standardfarbe
       hold = defaultColor;
       defaultColor = newColor;
       newColor = hold;
 
       // Funktion für die Übergangsberechnung des Bogens
       function arcTween(transition, newAngle) {
           transition.attrTween("d", function (d) {
               var interpolate = d3.interpolate(d.endAngle, newAngle);
               return function (t) {
                   d.endAngle = interpolate(t);
                   return arc(d);
               };
           });
       }
   };
 
   // Methode zur Umrechnung eines Werts in Grad
   this.valueToDegrees = function (value) {
       return Math.floor(value / this.config.max * 360);
   };
 
   // Methode zur Umrechnung eines Werts in Bogenmaß
   this.valueToRadians = function (value) {
       return this.valueToDegrees(value) * Math.PI / 180;
   };
 
   // Konfiguriert das Messgerät
   this.configure(configuration);
}
